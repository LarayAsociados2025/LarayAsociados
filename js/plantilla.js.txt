  // Cargar encabezado con título dinámico
  const encabezado = document.getElementById("encabezado");
  const tituloPagina = document.title;

    encabezado.innerHTML = `
      <div class="titulo-principal">${tituloPagina}</div>
    `;
  }

  // Cargar pie de página con botonera
  const footer = document.getElementById("footer");

    footer.innerHTML = `
      <div class="footer">
        <a href="../index.html">Página de Inicio</a>
        <a href="nuestra-firma.html">Nuestra Firma</a>
